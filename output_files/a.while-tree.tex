		\justifies

		\justifies
			\langle $if $z <= x \land !(x <= z \land z <= x)$ then $x := 0$ else $$skip$, s_{1,65}\rangle \rightarrow s_{1,65}
		\using
			[if_{ns}^{ff}]
		\end{prooftree}

	\using
		[comp_{ns}]
	\end{prooftree}
